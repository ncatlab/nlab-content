  \phantom{AA}

\[

    d G_7 
      & =
    -
    G_4 \wedge G_4
    +
    \tfrac{1}{4} p_2(N_X Q_{M5})
    \\
    & =
    -
    G_4 \wedge G_4

    \left( \tfrac{1}{4}p_1(N_X Q_{M5}) \right)

    \left( \tfrac{1}{4}p_1(N_X Q_{M5}) \right)

    \tfrac{1}{2} \chi(N_X Q_{M5})

    -
    \left(
      G_4 - \left( \tfrac{1}{4}p_1(N_X Q_{M5}) \right)

    \tfrac{1}{2} \chi(N_X Q_{M5})        
  \end{aligned}
\]

$$
  N_X Q_{M2}

  N_{Q_{M5}} Q_{M2}

$$

    \tfrac{1}{2}

  \end{aligned}

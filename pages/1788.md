
$$
  \begin{aligned}
    \frac{
      e^{i k_\mu (x^\mu - y^\mu)}
    }{
      - k_\mu k^\mu - \left( \tfrac{m c}{\hbar} \right)^2
    }
    & =
    \int 
      e^{
         i k_\mu(x^\mu - y^\mu) 
         +
         \tau 
         \left( 
            k_\mu k^\mu + \left( \tfrac{m c}{\hbar} \right)^2
         \right) 
      }
    \, d \tau
    \\
    & =
    \int 
      \exp\left(
         \tau
         \left(
            k^\mu + \tfrac{i}{2 \tau} (x^\mu - y^\mu)
         \right)
         \left(
            k_\mu + \tfrac{i}{2 \tau} (x_\mu - y_\mu)
         \right)
         +
         \tfrac{1}{4 \tau^2} (x^\mu - y^\mu)(x_\mu - y_\mu)
      \right)
      \, d \tau
  \end{aligned}
$$

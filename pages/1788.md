  (\Gamma_0^{\dot \alpha \alpha})
    \coloneqq
   \left[
      \array{
        1 & 0
        \\
        0 & -1
      }
   \right]
   \;\;\,,
   \;\;\;\;
  (\Gamma_1^{\dot \alpha \alpha})
    \coloneqq
   \left[
      \array{
        1 & 0
        \\
        0 & 1
      }
   \right]
   \;\;\,,
   \;\;\;\;
  (\Gamma_{n+1}^{\dot \alpha \alpha})
    \coloneqq
   \left[
      \array{
        0 & e_n
        \\
        \overline{e}_n & 0
      }
   \right]
$$

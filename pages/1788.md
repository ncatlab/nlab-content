    \mathbb{S}_{seq} Free_{seq}Mod( 
       \mathbb{S}_{seq} \otimes_{Day} y_{k_2}
       ,
       \mathbb{S}_{seq} \otimes_{Day} y_{k_1}
    )
    &  
    \simeq
    \overset{n}{\int} Seq(n + k_1 , k_2) \wedge \mathbb{S}_{seq}(n)
    \\
    &
    \simeq
    \left\{
      \array{
        S^{k_2-k_1} & if \; k_2 \geq k_1
        \\
        \ast & otherwise
      }
    \right.
  \end{aligned}
  \,.


$$
  \begin{aligned}
    \Delta_\pm(y-x)
     & =
     \left\{
       \array{
          \frac{i}{(2\pi)^{p}}
          \int
           \frac{1}{2\omega(\vec k)/c}
            \left(
              e^{-i \omega(\vec k)(x^0 - y^0)/c +  i \vec k \cdot (\vec x -\vec y)}
              -
              e^{+i \omega(\vec k)(x^0 - y^0)/c  + i \vec k \cdot (\vec x - \vec  y) }
            \right)
          d^p \vec k
          & \vert & \text{if} \,  \mp (x^0 - y^0) \gt 0
          \\
          0 & \vert & \text{otherwise}
       }
     \right.
     \\
     & =
     - \Delta_{\mp}(x-y)
  \end{aligned}
$$


$$
  R^{a_1 a_2}
  \;=\;
  \tfrac
    { \mathrm{R} }
    { D(D-1)/2 }
  \,
  e^{a_1}\, e^{a_2}
  \,,
  \;\;\;\;
  \text{hence}
  \;\;
  R^{a_1 a_2}{}_{b_1 b_2}
  \;=\;
  \tfrac
    { 2 k }
    { D (D-1) }
  \delta^{a_1 a_2}_{b_1 b_2}
  \,.
$$

$$
  \mathrm{Ric}_{a b}
  \;\equiv\;
  \eta_{a a'}
  R^{a' c}{}_{b c}
  \;=\;
    \tfrac
      { 2 \, \mathrm{R} }
      { D (D-1) }  
  \,
  \eta_{a a'}
  \underset{
    \tfrac{D-1}{2} \delta^{a'}_b
  }{
    \underbrace{
      \delta^{a' \, c}_{b \, c}
    }
  }
  \;=\;
  \tfrac
    { \mathrm{R} }
    { D }
  \,
  \eta_{a b}
$$

$$
  \mathrm{R}
  \;=\;
  \eta^{a b}
  \mathrm{Ric}_{a b}
$$

$$
  \mathrm{G}_{a b}
  \;\equiv\;
  \mathrm{Ric}_{a b}
  -
  \tfrac{1}{2}
  \mathrm{R}
  \, 
  \eta_{a b}
  \;=\;
  \big(
    \tfrac
      {1}
      {D}
      -
      \tfrac{1}{2}
  \big)
  \,
  \mathrm{R}
  \,
  \eta_{a b}
  \;=\;
  -
  \tfrac
    { D - 2 }
    { 2 D }
  \,
  \mathrm{R}
  \;
  \eta_{a b}
$$

$$
  \mathrm{G}_{a b}
  \;=\;
  \Lambda \, \eta_{a b}
  \;\;\;\;\;\;\;\;
  \Leftrightarrow
  \;\;\;\;\;\;\;\;
  \mathrm{R}
  \;=\;
  -
  \tfrac{2 D}{D-2}
  \, 
  \Lambda
$$

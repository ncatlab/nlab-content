  \exp(t \overline{\theta} Q )
  \overline{\theta} Q
  (E - \Omega)

  \exp(t \overline{\theta} Q )
  (\dot E - \dot \Omega)

  \mathrm{d}
  \big(
    \overline{\theta} Q

    \overline{\theta} Q
   \exp(\overline{\theta}Q)

  \exp(t \overline{\theta} Q )

  (\dot E - \dot \Omega)


$$
  (a,b,n)\cdot(a',b',n') 
   \;=\; 
  (a + a', b + b', n + n' + a b' - a' b)
$$

$$
  \begin{array}{l}
  \big((a,b,n)\cdot(a',b',n')\big) \cdot (a'', b'', c'')
  \\
  \;=\;
  (a + a', b + b', n + n' + a b' - a' b) \cdot (a'', b'', c'')
  \\
  \;=\;
  (-,-,n + n' + n'' + a b' - a' b + (a + a')b'' - a''(b + b')) 
  \\
  \;=\;
  \cdots
  a b' - a' b + a b '' + a' b'' - a'' b  - a'' b'
  \end{array}
$$

$$
  \begin{array}{l}
    (a,b,n)\cdot
    \big(
      (a',b',n') \cdot (a'', b'', c'')
    \big)
    \\
    \;=\;
    (a,b,n) \cdot (... a' b'' - a'' b')
    \\
    \;=\;
    a(b' + b'') - (a' + a'')b + a' b'' - a'' b'
    \\
    \;=\;
    a b' + a b'' - a' b - a'' b + a' b'' - a'' b'
  \end{array}
$$


$$
  (a,b,n)\cdot (-a, -b, -n) = (0,0,0)
$$

$$
  \begin{array}{l}
    (1,0,0)\cdot (0,1,0) \cdot (-1,0,0) \cdot (0,-1,0)
    \\
    \;=\;
    (1,1,1) \cdot (-1,0,0) \cdot (0,-1,0)
    \\
    \;=\;
    (0,1,1+1)  \cdot (0,-1,0)
    \\
    \;=\;
    (0,0,2)
  \end{array}
$$

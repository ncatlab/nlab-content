  DK(V_\bullet)\times DK(W_\bullet)

  \stackrel{\simeq}{\to}

  \stackrel{U(p)}{\to}
  U(\Gamma(V_\bullet)\otimes \Gamma(W_\bullet))
  \stackrel{U(\gamma)}{\longrightarrow}
  U(\Gamma(V_\bullet \otimes W_\bullet))
  \stackrel{U(\Gamma(f))}{\to}
  U(\Gamma(Z_\bullet))
  =
  DK(Z_\bullet)
  \,.

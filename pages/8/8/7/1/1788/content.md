

$$
  \begin{aligned}
   &
   \underset{
     \mathclap{
       x_1 \geq \cdots \geq x_N
     }
   }{\int}
    \;\;\;\;\;
   \underset{i \lt j}{\prod}
   \big(
     x_i - x_j 
   \big)
   e^{ - N \left\vert x\right\vert^2 /2 }
   \,
   d x_1 \cdots d x_{N}
   \\
   & 
   \;=\; 
   \underset{
     \mathclap{
       { x_1 \geq \cdots \geq x_N }
       \atop
       { {\vert x \vert}^2 = 1 }
     }
   }{\int}
    \;\;\;\;\;
   \underset{i \lt j}{\prod}
   \big(
     x_i - x_j 
   \big)
   dvol_{S^{N}}
   \,
   const N^{ 1/4 N(N -1) }
  \end{aligned}
$$

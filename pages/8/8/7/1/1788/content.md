
$$
  R^{a_1 a_2}
  \;=\;
  \tfrac
    {2 k}
    {D-1} 
  \, e^{a_1}\, e^{a_2}
$$

$$
  R^{a_1 a_2}{}_{b_1 b_2}
  \;=\;
  \tfrac
    {2 k}
    {D-1} 
  \delta^{a_1 a_2}_{b_1 b_2}
$$

$$
  Ric_{ab}
  \;=\;
  k\,
  \eta_{a b} 
$$

$$
  \mathrm{R}
  \;=\;
  k \, D
$$

$$
  G_{a b}
  \;=\;
  k(1-D/2) \eta_{a b}
$$

$$
  G_{a b}
  \;=\;
  -
  \Lambda \, \eta_{ab}
  \;\;\;\;\;
  \Leftrightarrow
  \;\;\;\;\;
  k
  \;=\;
  \frac
    { \Lambda }
    { D/2 - 1 }
$$

$$
  \begin{array}{l}
    Ric_{a b}
    \;=\;
    cnst_1
    \, 
    \eta_{a b}
    \\
    Ric_{i j}
    \;=\;
    cnst_2
    \,
    \delta_{i j}
  \end{array}
$$

$$
  \mathrm{R}
  \;=\;
  s \cdot cnst_1 
  +
  (D-s) \cdot cnst_2
$$

$$
  \begin{array}{l}
    G_{a b}
    \;=\;
    \Big(
      \big(1 - \tfrac{s}{2}\big)
      cnst_1
      -
      \frac{D-s}{2}  cnst_2
    \Big)
    \,
    \eta_{a b}
    \\
    G_{i j}
    \;=\;
    \Big(
      -\tfrac{s}{2}  cnst_1
      +
      \big(
        1-
        \tfrac{D-s}{2} 
      \big)
      cnst_2
    \Big)
    \,
    \delta_{i j}
  \end{array}
$$

$$
  \begin{array}{l}
    T_{a b}
    \;=\;
    \underset{
      +\,\tfrac{(s-1)!}{2}
    }{
      \underbrace{
      \big(
        (s-1)!
        -
        \tfrac{s!}{2s}
      \big)
      }
    }
    \,
    \eta_{a b}
    \\
    T_{i j}
    \;=\;
    \underset{
      -\,\tfrac{(s-1)!}{2}
    }{
      \underbrace{
          -
          \tfrac{s!}{2s}
      }
    }
    \,
    \eta_{i j}
  \end{array}
$$


$$
  \begin{array}{rcl}
      \big(1 - \tfrac{s}{2}\big)
      cnst_1
      -
      \frac{D-s}{2}  cnst_2
    &=&
     + \tfrac{(s-1)!}{2}
    \\
      -\tfrac{s}{2}  cnst_1
      +
      \big(
        1-
        \tfrac{D-s}{2} 
      \big)
      cnst_2
    &=&
     - \tfrac{(s-1)!}{2}
  \end{array}
$$

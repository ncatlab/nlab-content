
$$
  \mathrm{d}s^2
  \;\equiv\;
  -
  \mathrm{d}x^-\, \mathrm{d}x^+ 
  -
  \big(
    (\mu/3)^2 x^i x_i
    +
    (\mu/6)^2 x^j x_j
  \big)
  \mathrm{d}x^+ \, \mathrm{d}x^+
  +
  \mathrm{d}x^i\, \mathrm{d}x^i
  +
  \mathrm{d}x^j \, \mathrm{d}x^j
  \,,
  \;\;\;\;
  \begin{array}{l}
    i \in \{1, 2, 3\}
    \\
    j \in \{4,5,6,7,8,9\}
  \end{array}
$$

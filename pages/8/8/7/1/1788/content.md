  (s.o.): \;\;\; x^{a_1} x^{a_2} = + x^{a_2} x^{a_1}
$$

$$
  (s.o.): \;\;\;x^a \theta^\alpha = + \theta^\alpha x^a
$$

$$
  (s.o.): \;\;\;\theta^{\alpha_1} \theta^{\alpha_2}
  =
  +
  \theta^{\alpha_2} \theta^{\alpha_1}

  (s.o.): \;\;\;x^{a_1} (\mathbf{d}x^{a_2}) = + (\mathbf{d}x^{a_2}) x^{a_1}


$$
  \begin{array}{l}
    \mathrm{d} B^{a_1 a_2}
    -
    \omega^{a_1 a_2}{}_{a'_1 a'_2} \, B^{a'_1 a'_2}
    \;=\;
    \mathrm{d} B^{a_1 a_2}
    -\omega^{a_1}{}_{a'_1} B^{a'_1 a_2}
    -\omega^{a_2}{}_{a'_2} B^{a_1 a'_2}    
  \end{array}
$$

$$
  \omega^{a_1 a_2}{}_{a'_1 a'_2}
  \;=\;
  2
   \omega^{[a_1}{}_{[a'_1}
   \delta^{a_2]}_{a'_2]}
$$

$$
  \begin{array}{l}
    2
     \omega^{[a_1}{}_{[a'_1}
     \delta^{a_2]}_{a'_2]}
    B^{a'_1 a'_2}
    \\
    \;=\;
    2
     \omega^{[a_1}{}_{a'_1}
     \delta^{a_2]}_{a'_2}
    B^{a'_1 a'_2}    
    \\
    \;=\;
     \omega^{a_1}{}_{a'_1}
     \delta^{a_2}_{a'_2}
    B^{a'_1 a'_2}    
    -
    \underset{
     \omega^{a_2}{}_{a'_1}
     \delta^{a_1}_{a'_2}
    }{
    \underbrace{
     \omega^{a_2}{}_{a'_2}
     \delta^{a_1}_{a'_1}
    B^{a'_2 a'_1}    
    }
    }
    \\
    \;=\;
    \;=\;
     \omega^{a_1}{}_{a'_1}
     \delta^{a_2}_{a'_2}
    B^{a'_1 a'_2}    
    +
     \omega^{a_2}{}_{a'_2}
     \delta^{a_1}_{a'_1}
    B^{a'_1 a'_2}    
  \end{array}
$$

$$
  e_{a_1 a_2}
  \;=\;
  e^{a} B_{a,  a_1 a_2}
$$

$$
  \begin{array}{l}
    \mathrm{d}^\omega
    e_{a_1 a_2}
    \\
    \;=\;
    \mathrm{d}^\omega\big(e^{a} B_{a,  a_1 a_2}\big)
    \\
    \;=\;
    e^{a} e^{b} (\nabla_{b} B_{a, a_1 a_2})
  \end{array}
$$

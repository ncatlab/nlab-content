
$$
  \big[
    \vec g
    ,\,
    (\vec g', h) 
  \big]
  \;\mapsto\;
  \Big(
    [\vec g]
    ,\,
    \vec g^{-1} \cdot ( h \cdot \vec g' )
  \Big)
$$

$$
  \big(
    [\vec g]
    ,\,
    \vec g'
  \big)
  \;\mapsto\;
  \Big[
    \vec g
    ,\,
    \big(
      \big( 
        g_0^{-1}
        \cdot
        \vec g 
      \big)
        \cdot 
      \vec g'
      ,\,
      g_0
    \big)
  \Big]
$$

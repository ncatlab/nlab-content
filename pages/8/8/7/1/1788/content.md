
$$
  s\mathbf{C}(X,Y)_\bullet
  \;\coloneqq\;
  Hom_{s\mathbf{C}}\big(
    X \cdot \Delta[\bullet]
    ,\,
    Y
  \big)
$$

$$
  (X \otimes Y)_n
  \;\coloneqq\;
  X_ n \otimes Y_n
  \,.
$$

$$
  \array{
  Hom_{s\mathbf{C}}\big(
    X \cdot \Delta[n]
    ,\,
    Y
  \big)
  \times
  Hom_{s\mathbf{C}}\big(
    X' \cdot \Delta[n]
    ,\,
    Y'
  \big)
  &\longrightarrow&
  Hom_{s\mathbf{C}}\big(
    X \otimes X' \cdot \Delta[n]
    ,\,
    Y \otimes Y'
  \big)
  }
$$
